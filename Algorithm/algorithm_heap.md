    if(heapSize == 0) 
        return 0;
    
    int item = maxHeap[1]; 		
    maxHeap[1] = maxHeap[heapSize]; 	
    maxHeap[heapSize--] = 0; 		
    
    for(int i = 1; i*2 <= heapSize;) {
        
        if(maxHeap[i] > maxHeap[i*2] && maxHeap[i] > maxHeap[i*2+1]) {
            break;
        }
        else if (maxHeap[i*2] > maxHeap[i*2+1]) {
            swap(i, i*2);
            i = i*2;
        }
        else {
            swap(i, i*2+1);
            i = i*2+1;
        }
    }
    
    return item;
    
}
